204/** @internal @This checks if the buffer pipe state has changed.
205 * If the buffer pipe state has changed then it throws an update event
206 * (see @ref upipe_buffer_throw_update).
207 *
208 * @param upipe description structure of the pipe
209 */
210static void upipe_buffer_update(struct upipe *upipe)
211{
212 struct upipe_buffer *upipe_buffer = upipe_buffer_from_upipe(upipe);
213 enum upipe_buffer_state old_state = upipe_buffer->state;
214 enum upipe_buffer_state new_state = old_state;

215
216 if (upipe_buffer->low_limit &&
217 upipe_buffer->size < upipe_buffer->low_limit)
218 new_state = UPIPE_BUFFER_LOW;
219 else if (upipe_buffer->high_limit &&
220 upipe_buffer->size >= upipe_buffer->high_limit)
221 new_state = UPIPE_BUFFER_HIGH;
222 else
223 new_state = UPIPE_BUFFER_MIDDLE;
224
225 if (new_state != old_state) {
226 upipe_buffer->state = new_state;
227 upipe_buffer_throw_update(upipe, old_state, new_state);
228 }

230 uint64_t duration;
231 if (ubase_check(upipe_buffer_get_duration(upipe, &duration)))
232 upipe_verbose_va(upipe, "duration: %zums",
233 duration / (UCLOCK_FREQ27000000UL / 1000));

291/** @internal @This processes an input uref.
292 *
293 * @param upipe description structure of the pipe
294 * @param uref input uref
295 * @param upump_p reference to pump that generated the buffer
296 * @return true if the uref has been processed, false otherwise
297 */
298static bool_Bool upipe_buffer_process(struct upipe *upipe,
299 struct uref *uref,
300 struct upump **upump_p)
301{
302 struct upipe_buffer *upipe_buffer = upipe_buffer_from_upipe(upipe);
303 int ret;
304
305 size_t block_size;
306 ret = uref_block_size(uref, &block_size);
307 if (!ubase_check(ret)) {
308 upipe_err(upipe, "fail to get block size, dropping...")uprobe_log((upipe)->uprobe, upipe, UPROBE_LOG_ERROR, "fail to get block size, dropping..."
);
309 uref_free(uref);
310 return true1;
311 }
312
313 if (block_size + upipe_buffer->size > upipe_buffer->max_size)
314 return false0;
315
316 ret = upipe_buffer_upump_check(upipe);
317 if (!ubase_check(ret)) {
318 upipe_err(upipe, "fail to allocate upump, dropping...")uprobe_log((upipe)->uprobe, upipe, UPROBE_LOG_ERROR, "fail to allocate upump, dropping..."
);
319 uref_free(uref);
320 return true1;
321 }
322
323 uint64_t date;
324 if (ubase_check(uref_clock_get_dts_prog(uref, &date)) ||
325 ubase_check(uref_clock_get_dts_sys(uref, &date))) {
326 if (date < upipe_buffer->last_dts)
327 upipe_warn(upipe, "uref DTS is in the past")uprobe_log((upipe)->uprobe, upipe, UPROBE_LOG_WARNING, "uref DTS is in the past"
);
328 upipe_buffer->last_dts = date;
329 }

331 upipe_warn(upipe, "uref has no DTS")uprobe_log((upipe)->uprobe, upipe, UPROBE_LOG_WARNING, "uref has no DTS"
);
332
333 upipe_buffer->size += block_size;
334 ulist_add(&upipe_buffer->buffered, uref_to_uchain(uref));
335 upipe_buffer_update(upipe);

337}
