711static int ebur128_gated_loudness(ebur128_state** sts, size_t size,
712 double* out) {

715 double gated_loudness = 0.0;
716 size_t above_thresh_counter = 0;
717 size_t i, j, start_index;
718
719 for (i = 0; i < size; i++) {
720 if (sts[i] && (sts[i]->mode & EBUR128_MODE_I) != EBUR128_MODE_I) {
721 return EBUR128_ERROR_INVALID_MODE;

723 }

726 if (!sts[i]) continue;
727 if (sts[i]->d->use_histogram) {
728 for (j = 0; j < 1000; ++j) {

730 histogram_energies[j];

733 } else {

742 return EBUR128_SUCCESS;
743 }
744 relative_threshold /= (double) above_thresh_counter;
745 relative_threshold *= relative_gate_factor;
746 above_thresh_counter = 0;

748 start_index = 0;
749 } else {
