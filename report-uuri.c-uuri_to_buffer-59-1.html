370 userinfo = ustring_null();
371 tmp = *str;
372 }
373
374 struct ustring host = uuri_parse_host(&tmp);
375 if (ustring_is_null(host))
376 return authority;
377
378 if (ustring_is_null(ustring_split_match_str(&tmp, ":"))) {
379 *str = tmp;
380 authority.userinfo = userinfo;
381 authority.host = host;
382 return authority;
383 }
384
385 struct ustring port = uuri_parse_port(&tmp);
386 if (ustring_is_null(port))
387 return uuri_authority_null();
388 *str = tmp;
389 authority.userinfo = userinfo;
390 authority.host = host;
391 authority.port = port;
392 return authority;
393}

395struct ustring uuri_parse_path(struct ustring *str)
396{
397 struct ustring tmp = *str;
398 while (!ustring_is_empty(ustring_split_while(&tmp, uuri_path_set)) ||
399 !ustring_is_empty(ustring_split_pct_encoded(&tmp)));
400 struct ustring path = ustring_truncate(*str, str->len - tmp.len);
401 *str = tmp;
402 return path;
403}

405struct ustring uuri_parse_query(struct ustring *str)
406{
407 return ustring_split_while(str, uuri_query_set);
408}
409
410struct ustring uuri_parse_fragment(struct ustring *str)
411{
412 return ustring_split_while(str, uuri_fragment_set);
413}
414
415struct uuri uuri_parse(struct ustring *str)
416{
417 struct ustring tmp = *str;
418
419 struct uuri uuri = uuri_null();
420 uuri.scheme = uuri_parse_scheme(&tmp);
421 if (ustring_is_null(uuri.scheme) || !uuri.scheme.len)
422 return uuri_null();
423
424 if (ustring_is_null(ustring_split_match_str(&tmp, ":")))
425 return uuri_null();
426
427 if (!ustring_is_null(ustring_split_match_str(&tmp, "//"))) {
428 uuri.authority = uuri_parse_authority(&tmp);
429 if (uuri_authority_is_null(uuri.authority))
430 return uuri_null();
431 }
432
433 if (uuri_authority_is_null(uuri.authority) || ustring_match_str(tmp, "/"))
434 uuri.path = uuri_parse_path(&tmp);
435
436 if (!ustring_is_null(ustring_split_match_str(&tmp, "?")))
437 uuri.query = uuri_parse_query(&tmp);
438
439 if (!ustring_is_null(ustring_split_match_str(&tmp, "#")))
440 uuri.fragment = uuri_parse_fragment(&tmp);
441
442 *str = tmp;
443 return uuri;
444}

446int uuri_authority_len(const struct uuri_authority *authority, size_t *len_p)
447{
448 size_t len = 0;
449 if (authority->userinfo.len)
450 len += authority->userinfo.len + 1;
451 len += authority->host.len;
452 if (authority->port.len)
453 len += 1 + authority->port.len;
454 *len_p = len;
455 return UBASE_ERR_NONE;
456}

458int uuri_len(const struct uuri *uuri, size_t *len_p)
459{
460 size_t len = 0;
461
462 if (uuri->scheme.len)
463 len += uuri->scheme.len + 1;
464
465 if (!uuri_authority_is_null(uuri->authority)) {
466 size_t authority_len;
467 int ret = uuri_authority_len(&uuri->authority, &authority_len);
468 if (!ubase_check(ret))
469 return ret;
470
471 len += 2 + authority_len;
472 }
473
474 if (uuri->path.at)
475 len += uuri->path.len;
476 if (uuri->query.len)
477 len += 1 + uuri->query.len;
478 if (uuri->fragment.len)
479 len += 1 + uuri->fragment.len;
480 *len_p = len;
481 return UBASE_ERR_NONE;
482}

484int uuri_authority_to_buffer(const struct uuri_authority *authority,
485 char *buffer, size_t len)
486{
487 int ret;
488
489 if (uuri_authority_is_null(*authority))
490 return UBASE_ERR_INVALID;
491
492 if (authority->userinfo.len) {
493 ret = snprintf(buffer, len, "%.*s@",
494 (int)authority->userinfo.len,
495 authority->userinfo.at);
496 if (ret < 0 || ret >= len)
497 return UBASE_ERR_INVALID;
498 buffer += ret;
499 len -= ret;
500 }
501
502 ret = snprintf(buffer, len, "%.*s",
503 (int)authority->host.len,
504 authority->host.at);
505 if (ret < 0 || ret >= len)
506 return UBASE_ERR_INVALID;
507 buffer += ret;
508 len -= ret;
509
510 if (authority->port.len) {
511 ret = snprintf(buffer, len, ":%.*s",
512 (int)authority->port.len,
513 authority->port.at);
514 if (ret < 0 || ret >= len)
515 return UBASE_ERR_INVALID;
516 buffer += ret;
517 len -= ret;
518 }
519
520 return UBASE_ERR_NONE;
521}

523int uuri_to_buffer(struct uuri *uuri, char *buffer, size_t len)
524{
525 int ret;
526
527 if (uuri_is_null(*uuri))
528 return UBASE_ERR_INVALID;
529
530 ret = snprintf(buffer, len, "%.*s:", (int)uuri->scheme.len,
531 uuri->scheme.at);
532 if (ret < 0 || ret >= len)
533 return UBASE_ERR_INVALID;
534 buffer += ret;
535 len -= ret;
536
537 if (!uuri_authority_is_null(uuri->authority)) {
538 ret = snprintf(buffer, len, "//");
539 if (ret < 0 || ret >= len)
540 return UBASE_ERR_INVALID;
541 buffer += ret;
542 len -= ret;
543
544 size_t authority_len;
545 ret = uuri_authority_len(&uuri->authority, &authority_len);
546 if (!ubase_check(ret))
547 return ret;
548
549 if (authority_len >= len)
550 return UBASE_ERR_NOSPC;
551
552 ret = uuri_authority_to_buffer(&uuri->authority, buffer,
553 authority_len + 1);
554 if (!ubase_check(ret))
555 return ret;
556
557 buffer += authority_len;
558 len -= authority_len;
559 }
560
561 ret = snprintf(buffer, len, "%.*s",
562 (int)uuri->path.len,
563 uuri->path.at);
564 if (ret < 0 || ret >= len)
565 return UBASE_ERR_INVALID;
566 buffer += ret;
567 len -= ret;
568
569 if (uuri->query.len) {
570 ret = snprintf(buffer, len, "?%.*s", (int)uuri->query.len,
571 uuri->query.at);
572 if (ret < 0 || ret >= len)
573 return UBASE_ERR_INVALID;
574 buffer += ret;
575 len -= ret;
576 }
577 if (uuri->fragment.len) {
578 ret = snprintf(buffer, len, "#%.*s", (int)uuri->fragment.len,
579 uuri->fragment.at);
580 if (ret < 0 || ret >= len)
581 return UBASE_ERR_INVALID;
582 buffer += ret;
583 len -= ret;
Value stored to 'len' is never read
584 }
585
586 return UBASE_ERR_NONE;
587}

589int uuri_to_str(struct uuri *uuri, char **str_p)
590{
591 size_t size = 0;
592 int ret = uuri_len(uuri, &size);
593 if (ret)
594 return ret;
595
596 char *str = malloc((size + 1) * sizeof (char));
597 if (!str)
598 return UBASE_ERR_ALLOC;
599
600 ret = uuri_to_buffer(uuri, str, (size + 1) * sizeof (char));
601 if (ret) {
602 free(str);
603 return ret;
604 }
605
606 *str_p = str;
607 return UBASE_ERR_NONE;
