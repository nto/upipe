115static bool_Bool uprobe_selflow_check_def(struct uprobe *uprobe, const char *def)
116{
117 struct uprobe_selflow *uprobe_selflow = uprobe_selflow_from_uprobe(uprobe);
118 switch (uprobe_selflow->type) {
119 case UPROBE_SELFLOW_VOID:
120 return !ubase_ncmp(def, "void.");
121 case UPROBE_SELFLOW_PIC: {
122 const char *pos;
123 return (!ubase_ncmp(def, "pic.") && ubase_ncmp(def, "pic.sub.")) ||
124 ((pos = strstr(def, ".pic.")) != NULL((void*)0) &&
125 ubase_ncmp(pos, ".pic.sub."));
126 }
127 case UPROBE_SELFLOW_SOUND:
128 return !ubase_ncmp(def, "sound.") || strstr(def, ".sound.") != NULL((void*)0);
129 case UPROBE_SELFLOW_SUBPIC:
130 return !ubase_ncmp(def, "pic.sub.") ||
131 strstr(def, ".pic.sub.") != NULL((void*)0);
132 default:
133 return false0;
134 }
135}

144static bool_Bool uprobe_selflow_check(struct uprobe *uprobe, uint64_t flow_id,
145 struct uref *flow_def)
146{
147 struct uprobe_selflow *uprobe_selflow = uprobe_selflow_from_uprobe(uprobe);
148 if (!strcmp(uprobe_selflow->flows, "all") ||

152 /* comma-separated list of subs */

154 uint64_t found;
155 char attr[strlen(flows) + 1];
156 char value[strlen(flows) + 1];
157 int consumed;
158 int tmp;
159 while ((tmp = sscanf(flows, "%"PRIu64"l" "u"",%n", &found, &consumed)) >= 1 ||
160 sscanf(flows, "%[a-zA-Z]=%[^,],%n", attr, value,
161 &consumed) >= 2) {
162 flows += consumed;
163 if (tmp > 0) {
164 if (found == flow_id)
165 return true1;
166 } else if (!strcmp(attr, "lang")) {
167 uint8_t languages;
168 if (ubase_check(uref_flow_get_languages(flow_def, &languages)) &&
169 languages) {
170 for (uint8_t j = 0; j < languages; j++) {
171 const char *lang;
172 if (ubase_check(uref_flow_get_language(flow_def, &lang,
173 j)) &&
174 !strcmp(lang, value))
175 return true1;
176 }
177 }
178 } else if (!strcmp(attr, "name")) {
179 const char *name;
180 if (ubase_check(uref_program_flow_get_name(flow_def, &name)) &&
